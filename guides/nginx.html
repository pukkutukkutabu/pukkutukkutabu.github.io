# NGINX

# PFX формат сертификатов

PFX - формат файла, в котором запакованы доменный сетификат, приватный ключ и цепочки сетификатов (промежуточный и корневой). Для того чтобы он был понятен для nginx, из него необходимо достатать эти сертификаты в понятной для nginx форме

# Настройка SSL и распаковка сертификатов из формата

# Распаковать доменный сертификат

openssl pkcs12 -in ./cert.pfx -clcerts -nokeys -out domain.crt

# Распаковать приватный ключ

openssl pkcs12 -in ./cert.pfx -nocerts -nodes -out domain.key

# Рапспаковать цепочку сертификатов (промежуточный и корневой)

openssl pkcs12 -in ./cert.pfx -cacerts -nokeys -out ca.crt

# Настройка ssl внутри nginx

Перед использованием, часто необходимо склеивать сетефикаты и цепочку с промежуточным и корневым сертификатом в один bundle

echo "" >> domain.crt
cat domain.crt ca.crt > bundle.crt
cp domain.key /etc/nginx/ssl/
cp bundle.crt /etc/nginx/ssl/

Настройка осуществляется в блоке server. Строки необходимые для работы сертификата:

listen 443 ssl;

ssl_certificate      ssl/bundle.crt;
ssl_certificate_key  ssl/domain.key;
